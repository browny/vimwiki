class Result {
    int getFirst() { return 1; }
    int getLast() { return 9; }
}

interface calSum<T> {
    T sum(Result result);
}

class StringCalSum implements calSum<String> {
    @Override
    public String sum(Result result) {
        return Integer.toString(result.getFirst()) +
            Integer.toString(result.getLast());
    }
}

class IntCalSum implements calSum<Integer> {
    @Override
    public Integer sum(Result result) {
        return result.getFirst() + result.getLast();
    }
}

public class HelloWorld {

    // generics method
    public static <T> T calculateSum(Result result, calSum<T> cal) {
        T out = cal.sum(result);
        return out;
    }

    public static void main(String []args){

        Result result  = new Result();

        String sum1 = Integer.toString(result.getFirst()) +
            Integer.toString(result.getLast());
        System.out.println(sum1);

        int sum2 = result.getFirst() + result.getLast();
        System.out.println(Integer.toString(sum2));


        StringCalSum s = new StringCalSum();
        String sum3 = HelloWorld.calculateSum(result, s);
        System.out.println(sum3);

        IntCalSum i = new IntCalSum();
        int sum4 = HelloWorld.calculateSum(result, i);
        System.out.println(Integer.toString(sum4));


    }
}
