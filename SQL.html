MySQL commands

$ mysql -u <userName> -p -h <ip-address>

mysql> show databases;

mysql> use [db name];               // switch to a database

mysql> show tables;                 // show all tables

mysql> describe [table name];       // see database's field formats

mysql> SELECT * FROM [table name];  // show all data in a table

mysql> SELECT * FROM [table name] WHERE [field name] = "whatever";

mysqldump -u <user-name> -p <db-name> ooxx.sql // export db

mysql -u <user-name> -p -h <host-name> <db-name> < ooxx.sql // import db
